def python_test(n):
    a = [i**2 for i in range(n)]
    b = [i**3 for i in range(n)]
    c = []
    for i in range(n):
        c.append(a[i] + b[i]
    return c
python_test(10)
Out[28]:
[0, 2, 12, 36, 80, 150, 252, 392, 576, 810]

def numpy_test(n):
    a = np.arange(n) ** 2
    b = np.arange(n) ** 3
    c = a + b
    return c
numpy_test(10)
